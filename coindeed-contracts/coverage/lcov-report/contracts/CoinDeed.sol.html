    function claimBalance() onlyInState(ICoinDeedFactory.DeedState.CLOSED) external override {
        uint256 tokenAToClaim = totalReturn;
 
        // in case user is claiming balance before management fee is claimed
        uint256 fee = _managementFee();
        tokenAToClaim = tokenAToClaim.sub(fee).add(totalFeeClaimed);
 
        // buyer can claim tokenA in the same proportion as their buyins
        uint256 claimAmount = buyIns[msg.sender].mul(tokenAToClaim).div(totalSupply);
 
        // just a sanity check in case division rounds up
        if (claimAmount > totalReturn) {
            claimAmount = totalReturn;
        }
 
        // if user has a loan to pay, pay the loan by swapping in deed token
        // the _payoff function takes care of sending back left over deed token
        if (loans[msg.sender] > 0) {
            uint256 deedTokenAmount = _swapTokenToToken(pair.tokenA, claimAmount, address(deedToken));
            _payOff(deedTokenAmount);
        } else {
            IERC20(pair.tokenA).transfer(msg.sender, claimAmount);
        }
 
        // reset globals
        totalSupply = totalSupply.sub(buyIns[msg.sender]);
        buyIns[msg.sender] = 0;
        totalReturn = totalReturn.sub(claimAmount);
    }
